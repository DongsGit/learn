### 1.HelloWorld

~~~sql
create or replace procedure MYDEMO01 is
begin
  dbms_output.put_line('Hello World!');
end MYDEMO01;
~~~

### 2.存储过程调用

~~~sql
①declare
begin
  MYDEMO01;
end;
 
begin
  MYDEMO01; --在此处也可使用myDemo01();完成存储过程的调用
end;
②call MYDEMO01();--call 存储过程名可完成调用，注意括号不能少
③SQL> set serveroutput on --可以再command命令窗口执行
 exec MYDEMO01 --下面是输出结果
 hello word, my name is stored procedure
 PL/SQL procedure successfully completed

~~~

### 3.变量申明，赋值

~~~sql
create or replace procedure MYDEMO02
as
name varchar(10);--声明变量，注意varchar需要指定长度
age int;
begin
  name:='xiaoming';--变量赋值
  age:=18;
  dbms_output.put_line('name='||name||', age='||age);--通过||符号达到连接字符串的功能
end;
~~~

#### 4.带有参数的存储过程

~~~sql
create or replace procedure MYDEMO03(name in varchar,age in int)
as
begin
  dbms_output.put_line('name='||name||', age='||age);
end;
~~~

### 5.异常 问题

~~~sql
create or replace procedure mydemo04
as
age int;
begin
  age:=10/0;
  dbms_output.put_line(age);
  exception when others   then
    dbms_output.put_line('error');
end;
~~~

### 6.循环遍历查询结果并且打印

~~~sql
declare 
begin
for cryptic_rec in (
  select caption from STA_PR_REG
)
loop
dbms_output.put_line('名称：'||cryptic_rec.caption);
end loop;
end;
~~~
